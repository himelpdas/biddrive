#to ensure this process runs all the time, put this in crontab every 6 hours without locking on a shared server (because adminds can kill process)
cd ~/biddrive.com && killall python && nohup ~/local_bin/python/bin/python web2py.py -K init -X -a gbkyjh78u -i 127.0.0.1 -p 8888
#Note the local python binary, as this is where our custom python is, and will avoid import errors from using the system python, since crontab doesn't respect bash_rc:
##

NOTES- 2 programs at once
nohup works in shell but IS NOT RESPECTED in cron... cron reboot doesn't seem to work on dreamhost
program1 && program2 #program2 runs after program1, for log running processes they will not run at the same time.
program1 & program2 & #will run in parallel
killall python && program1 & program2 #will run program1 after [killall python], program2 will start with [killall python[, and will be killed immediately 
killall python && (program1 & program2 &) #will work fine, except when there are no python running
killall python ; (program1 & program2 &) #will work regardless if killall python succeeds or not (BEST)
